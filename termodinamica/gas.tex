\subsection{Trasformazioni isobare}

\begin{equation*}
	\frac{V}{T} = \frac{nR}{p}
\end{equation*}
da cui
\begin{equation}
	\frac{V_1}{T_1} = \frac{V_2}{T_2}
\end{equation}

Il lavoro Ã¨ calcolato come
\begin{equation*}
	\Delta U = n c_V \Delta T
\end{equation*}
da cui
\begin{equation*}
	L = p (V_B - V_A) = p \left(\frac{n R T_B}{p} - \frac{n R T_A}{p}\right)
\end{equation*}

\begin{equation}
	\highlight{L = n R (T_B - T_A)}
\end{equation}

Mentre il calore
\begin{equation}
	Q = n c_p (T_B - T_A)
\end{equation}
grazie alla relazione dei Mayer $c_p = c_V + R$
